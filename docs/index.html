fetch('https://demo.dotcms.com/api/v1/authentication/api-token', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json'
    },
    body: JSON.stringify({
        user: 'admin@dotcms.com',
        password: 'admin',
        expirationDays: 10
    })
})
    .then(data => data.json())
    .then(data => console.log(data.entity));
